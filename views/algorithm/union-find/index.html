class UnineFind {

function longestConsecutive(nums) {
  const len = nums.length
  const u = new UnineFind(len)
  // å­˜å‚¨å·²ç»è®¿é—®è¿‡çš„å…ƒç´ ï¼Œä»¥åŠå…¶ä¸‹æ ‡

    const item = nums[i]

    // è¿æ¥å‰ä¸€ä¸ª
    if (visited[item - 1] != null) {
      u.merge(i, visited[item - 1])
    }
    // è¿æ¥åä¸€ä¸ª

  }
